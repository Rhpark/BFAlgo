#include<iostream>
#include<vector>

using namespace std;

class Algo
{
	int total;
	int sum;
	int len[3];
	int *ary;

	vector <int> vclen;
	vector <int> vctotal;

public:
	void InputTotal()
	{
		cin>>total;
		for(int i = 0 ; i<total ; i ++)
		{
			InputLen();
			Cacul();
		}
		Print();
	}

	void InputLen()
	{	
		int max = 0;
		sum = 0;
		int val1,val2;
		for(int i=0 ; i < 3 ; i ++)
		{
			cin>>len[i];
			sum += len[i];
		}

		int *map = new int [sum*2];
		
		for(int i = 0 ; i < sum*2 ; i ++)
		{
			cin>>map[i]>>map[i+1];
			if(max < map[i+1])
				max = map[i+1];
			i++;
		}
		ary = new int[max];
		for(int i = 0 ; i < sum*2 ; i ++)
		{
			for(int j = map[i] ; j < map[i+1] ; j ++)
			{
				ary[j] = 1;
			}
			i++;
		}
	}
	void Cacul()
	{
		int cnt = 0;
		for(int i = 0 ; i <= 3000 ; i ++)
		{
			if(ary[i] == 1)
			{
				cnt++;
			}
			else break;
		}
		vctotal.push_back(cnt);
	}
	void Print()
	{
		for(int i = 0 ; i < vctotal.size() ; i ++)
		{
			cout<<vctotal[i]<<endl;
		}
	}
};

int main()
{
	Algo a;
	a.InputTotal();
	return 0;
}
